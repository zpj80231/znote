# JwtTokenUtilе·Ҙе…·зұ»

  • дё»иҰҒеҢ…жӢ¬з”ҹжҲҗtokenпјҢи®ҫзҪ®иҝҮжңҹж—¶й—ҙпјҢйӘҢиҜҒtokenжҳҜеҗҰжңүж•ҲпјҢиҺ·еҸ–tokenиҙҹиҪҪзӯү
  • е·Ҙе…·зұ»еҸҜе…·дҪ“еҸӮиҖғmallйЎ№зӣ®
package com.macro.mall.tiny.common.utils;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.stereotype.Component;

import java.util.Date;
import java.util.HashMap;
import java.util.Map;

/**
 * JwtTokenз”ҹжҲҗзҡ„е·Ҙе…·зұ»
 * Created by macro on 2018/4/26.
 */
@Component
public class JwtTokenUtil {
    private static final Logger LOGGER = LoggerFactory.getLogger(JwtTokenUtil.class);
    private static final String CLAIM_KEY_USERNAME = "sub";
    private static final String CLAIM_KEY_CREATED = "created";
    @Value("${jwt.secret}")
    private String secret;
    @Value("${jwt.expiration}")
    private Long expiration;

    /**
     * ж №жҚ®иҙҹиҙЈз”ҹжҲҗJWTзҡ„token
     */
    private String generateToken(Map<String, Object> claims) {
        return Jwts.builder()
                .setClaims(claims)
                .setExpiration(generateExpirationDate())
                .signWith(SignatureAlgorithm.HS512, secret)
                .compact();
    }

    /**
     * д»ҺtokenдёӯиҺ·еҸ–JWTдёӯзҡ„иҙҹиҪҪ
     */
    private Claims getClaimsFromToken(String token) {
        Claims claims = null;
        try {
            claims = Jwts.parser()
                    .setSigningKey(secret)
                    .parseClaimsJws(token)
                    .getBody();
        } catch (Exception e) {
            LOGGER.info("JWTж јејҸйӘҢиҜҒеӨұиҙҘ:{}",token);
        }
        return claims;
    }

    /**
     * з”ҹжҲҗtokenзҡ„иҝҮжңҹж—¶й—ҙ
     */
    private Date generateExpirationDate() {
        return new Date(System.currentTimeMillis() + expiration * 1000);
    }

    /**
     * д»ҺtokenдёӯиҺ·еҸ–зҷ»еҪ•з”ЁжҲ·еҗҚ
     */
    public String getUserNameFromToken(String token) {
        String username;
        try {
            Claims claims = getClaimsFromToken(token);
            username =  claims.getSubject();
        } catch (Exception e) {
            username = null;
        }
        return username;
    }

    /**
     * йӘҢиҜҒtokenжҳҜеҗҰиҝҳжңүж•Ҳ
     *
     * @param token       е®ўжҲ·з«Ҝдј е…Ҙзҡ„token
     * @param userDetails д»Һж•°жҚ®еә“дёӯжҹҘиҜўеҮәжқҘзҡ„з”ЁжҲ·дҝЎжҒҜ
     */
    public boolean validateToken(String token, UserDetails userDetails) {
        String username = getUserNameFromToken(token);
        return username.equals(userDetails.getUsername()) && !isTokenExpired(token);
    }

    /**
     * еҲӨж–ӯtokenжҳҜеҗҰе·Із»ҸеӨұж•Ҳ
     */
    private boolean isTokenExpired(String token) {
        Date expiredDate = getExpiredDateFromToken(token);
        return expiredDate.before(new Date());
    }

    /**
     * д»ҺtokenдёӯиҺ·еҸ–иҝҮжңҹж—¶й—ҙ
     */
    private Date getExpiredDateFromToken(String token) {
        Claims claims = getClaimsFromToken(token);
        return claims.getExpiration();
    }

    /**
     * ж №жҚ®з”ЁжҲ·дҝЎжҒҜз”ҹжҲҗtoken
     */
    public String generateToken(UserDetails userDetails) {
        Map<String, Object> claims = new HashMap<>();
        claims.put(CLAIM_KEY_USERNAME, userDetails.getUsername());
        claims.put(CLAIM_KEY_CREATED, new Date());
        return generateToken(claims);
    }

    /**
     * еҲӨж–ӯtokenжҳҜеҗҰеҸҜд»Ҙиў«еҲ·ж–°
     */
    public boolean canRefresh(String token) {
        return !isTokenExpired(token);
    }

    /**
     * еҲ·ж–°token
     */
    public String refreshToken(String token) {
        Claims claims = getClaimsFromToken(token);
        claims.put(CLAIM_KEY_CREATED, new Date());
        return generateToken(claims);
    }
}

# jwt+redisдҪҝз”ЁжөҒзЁӢ

  1. зҷ»еҪ•пјҡз”ЁжҲ·з¬¬дёҖж¬Ўзҷ»еҪ•пјҢж ЎйӘҢйҖҡиҝҮпјҢз”ҹжҲҗtokenе№¶еӯҳеҲ°redisйҮҢпјҢиҝ”еӣһе®ўжҲ·з«ҜпјҲtokenи®ҫзҪ®иҝҮжңҹж—¶й—ҙ1еӨ©пјҢredisи®ҫзҪ®иҝҮжңҹж—¶й—ҙ0.5е°Ҹж—¶пјүгҖӮ

  2. йүҙжқғпјҡд№ӢеҗҺе®ўжҲ·з«ҜжҜҸж¬ЎиҜ·жұӮйғҪжҗәеёҰtokenпјҢжңҚеҠЎз«Ҝж ЎйӘҢtokenйҖҡиҝҮе№¶дё”redisиҝҮжңҹж—¶й—ҙжІЎиҝҮпјҢеҲҷredisз»ӯжңҹз„¶еҗҺиҝ”еӣһtokenпјӣжңҚеҠЎз«Ҝж ЎйӘҢtokenйҖҡиҝҮдҪҶеҰӮжһңredisдёӯtokenе·ІиҝҮжңҹпјҢеҲҷеӨұж•ҲгҖӮ

  3. зҷ»еҮәпјҡжңҚеҠЎз«Ҝж ЎйӘҢtokenйҖҡиҝҮпјҢе°Ҷredisдёӯзҡ„tokenеҲ йҷӨгҖӮ

# е®үе…Ёй—®йўҳ

  1. еҰӮжһңе…¶д»–з”ЁжҲ·иҺ·еҫ—жҲ‘д»¬з”өи„‘дёҠзҡ„token,йӮЈд№Ҳд»–д»¬е°ұиғҪжЁЎд»ҝзңҹе®һз”ЁжҲ·иҝӣиЎҢж“ҚдҪңпјҲй»‘е®ўзӣ‘жҺ§з”өи„‘пјҢиҺ·еҫ—зҪ‘з«ҷеҸ‘йҖҒзҡ„token,иҝӣиЎҢж“ҚдҪңпјүпјҹ

    еҜ№дәҺж•Ҹж„ҹзҡ„apiжҺҘеҸЈпјҢйңҖдҪҝз”Ёhttps гҖӮhttpsжҳҜеңЁhttpи¶…ж–Үжң¬дј иҫ“еҚҸи®®еҠ е…ҘSSLеұӮпјҢе®ғеңЁзҪ‘з»ңй—ҙйҖҡдҝЎжҳҜеҠ еҜҶзҡ„пјҢжүҖд»ҘйңҖиҰҒеҠ еҜҶиҜҒд№ҰгҖӮ

  2. жҲ‘д»¬жҳҜеҗҰеҸҜд»ҘдјӘйҖ з”ЁжҲ·tokenиҝӣиЎҢи®ҝй—®пјҹ

    дёҚеҸҜд»ҘпјҢеӣ дёәдҪ ж— жі•дҪҝз”ЁжңҚеҠЎеҷЁзҡ„зӯҫеҗҚпјҢе°ұжҳҜдҪ иҮӘе·ұзҡ„еҜҶй’ҘзӯҫеҗҚзҡ„дҝЎжҒҜжңҚеҠЎеҷЁиҜҶеҲ«дёҚдәҶгҖӮ

  3. жҲ‘д»¬жҳҜеҗҰеҸҜд»Ҙдҝ®ж”№tokenдёӯbodyзҡ„дҝЎжҒҜпјҹ

    дёҚиғҪпјҢдҝ®ж”№дәҶд№ӢеҗҺзӯҫеҗҚдҝЎжҒҜе°ұдёҚжӯЈзЎ®пјҢз„¶еҗҺе°ұж— жі•йӘҢиҜҒзӯҫеҗҚпјҢиҜҙжҳҺж•°жҚ®иў«дҝ®ж”№дәҶгҖӮ

  4. appзҷ»еҪ•еҗҺпјҢжңҚеҠЎз«Ҝиҝ”еӣһдёҖдёӘtokenпјҢappеӯҳеңЁе®ўжҲ·з«ҜпјҢдёӢж¬ЎеҶҚжү“ејҖappж—¶пјҢзӣҙжҺҘиҜ»tokenпјҢдј tokenеҲ°жңҚеҠЎз«ҜеҒҡйӘҢиҜҒпјҢе…ҚеҺ»йҮҚж–°иҫ“е…Ҙз”ЁжҲ·еҜҶз Ғзҡ„йә»зғҰпјҢиҝҷдёӘtokenеӯҳеӮЁеңЁheaderйҮҢпјҢзӣ®еүҚзңӢеӨ§еӨҡж•°appйғҪжҳҜиҝҷж ·еҒҡпјҢдҪҶеҰӮжһңй»‘е®ўжҠ“еҢ…иҺ·еҸ–еҲ°tokenпјҢдјӘйҖ http иҜ·жұӮпјҢеҜ№жңҚеҠЎеҷЁеҒҡж“ҚдҪңпјҢйӮЈеІӮдёҚжҳҜеҫҲдёҚе®үе…ЁгҖӮгҖӮгҖӮ

    иҝҷж–№жі•зЎ®е®һдёҚеҘҪе•ҠпјҢдёҚиғҪеҸӘдҫқиө–дәҺhttpзҡ„headerйҮҢзҡ„дёңиҘҝжқҘи®ӨиҜҒпјҢеӨӘе®№жҳ“жЁЎд»ҝгҖӮжңҖз®ҖеҚ•зҡ„ж–№жі•еҸҜиғҪе°ұжҳҜиө°httpsпјҢе®ўжҲ·з«ҜеҸӘиҰҒжҺҘеҸ—жңҚеҠЎеҷЁз«Ҝзҡ„зӯҫеҗҚеҚіеҸҜгҖӮ

    зҗҶи§ЈпјҡзӯҫеҗҚе°ұжҳҜйӘҢиҜҒдҝЎжҒҜзҡ„е”ҜдёҖжҖ§пјҢеҰӮжһңдёӯй—ҙдәәиҝӣиЎҢиҺ·еҫ—tokenпјҢйӮЈд№Ҳж— жі•иҝӣиЎҢе…¬й’ҘзӯҫеҗҚпјҢжңҚеҠЎеҷЁиҺ·еҫ—tokenд№ӢеҗҺиҝҳиҰҒиҝӣиЎҢи§ЈеҜҶзҡ„пјҢеӣ жӯӨиҝҷдёӘж–№жі•еҸҜд»ҘиҝӣиЎҢйҒҝе…Қж”»еҮ»гҖӮ

# жҖ»з»“

  • дјҳзӮ№пјҡеңЁйқһи·ЁеҹҹзҺҜеўғдёӢдҪҝз”ЁJWTжңәеҲ¶жҳҜдёҖдёӘйқһеёёдёҚй”ҷзҡ„йҖүжӢ©пјҢе®һзҺ°ж–№ејҸз®ҖеҚ•пјҢж“ҚдҪңж–№дҫҝпјҢиғҪеӨҹеҝ«йҖҹе®һзҺ°гҖӮз”ұдәҺжңҚеҠЎз«ҜдёҚеӯҳеӮЁз”ЁжҲ·зҠ¶жҖҒдҝЎжҒҜпјҢеӣ жӯӨеӨ§з”ЁжҲ·йҮҸпјҢеҜ№еҗҺеҸ°жңҚеҠЎд№ҹдёҚдјҡйҖ жҲҗеҺӢеҠӣгҖӮ

  • зјәзӮ№пјҡи·Ёеҹҹе®һзҺ°зӣёеҜ№жҜ”иҫғйә»зғҰпјҢе®үе…ЁжҖ§д№ҹжңүеҫ…жҺўи®ЁгҖӮеӣ дёәJWTд»ӨзүҢиҝ”еӣһеҲ°йЎөйқўдёӯпјҢеҸҜд»ҘдҪҝз”ЁjsиҺ·еҸ–еҲ°пјҢеҰӮжһңйҒҮеҲ°XSSж”»еҮ»д»ӨзүҢеҸҜиғҪдјҡиў«зӣ—еҸ–пјҢеңЁJWTиҝҳжІЎи¶…ж—¶зҡ„жғ…еҶөдёӢпјҢе°ұдјҡиў«иҺ·еҸ–еҲ°ж•Ҹж„ҹж•°жҚ®дҝЎжҒҜгҖӮ

  • еҜ№дәҺе®үе…Ёй—®йўҳпјҡ

    еҜ№дәҺж•Ҹж„ҹзҡ„apiжҺҘеҸЈпјҢйңҖдҪҝз”Ёhttps гҖӮhttpsжҳҜеңЁhttpи¶…ж–Үжң¬дј иҫ“еҚҸи®®еҠ е…ҘSSLеұӮпјҢе®ғеңЁзҪ‘з»ңй—ҙйҖҡдҝЎжҳҜеҠ еҜҶзҡ„пјҢжүҖд»ҘйңҖиҰҒеҠ еҜҶиҜҒд№ҰгҖӮйҮҮз”Ёhttps жҲ–иҖ… д»Јз ҒеұӮйқўд№ҹеҸҜд»ҘеҒҡе®үе…ЁжЈҖжөӢпјҢжҜ”еҰӮipең°еқҖеҸ‘з”ҹеҸҳеҢ–пјҢMACең°еқҖеҸ‘з”ҹеҸҳеҢ–зӯүзӯүпјҢеҸҜд»ҘиҰҒжұӮйҮҚж–°зҷ»еҪ•
