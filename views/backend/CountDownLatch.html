# йҖӮз”ЁеңәжҷҜ

  • еёёе’ҢзәҝзЁӢжұ е…ұеҗҢеҮәзҺ°пјҢз”ЁеҒҡд»»еҠЎзҡ„еҲҶеүІеҗҲе№¶пјҢеҰӮпјҡ

    жҲ‘д»¬жңү90дёҮж•°жҚ®пјҢйңҖиҰҒеӨҡзәҝзЁӢеӨҡ Sheet йЎөеҜјеҮә ExcelпјҢиҝҷж—¶еҖҷе°ұйңҖиҰҒз”ЁеҲ° CountDownLatch жқҘе°ҶиҝҷдёҖд»»еҠЎе…·дҪ“з»ҶеҲҶгҖӮ

    1. е°ҶеҜјеҮә Excel зңӢдҪңдёҖйЎ№еӨ§д»»еҠЎгҖӮ

    2. з”ЁзәҝзЁӢжұ ж“ҚдҪңпјҢеҜјеҮәзҡ„жҜҸдёӘ Sheet йЎөдёәе…¶дёӯдёҖйЎ№е°Ҹд»»еҠЎгҖӮ

      2.1 е‘ҠиҜү CountDownLatch жҖ»д»»еҠЎж•°

      CountDownLatch countDownLatch = new CountDownLatch(д»»еҠЎжҖ»ж•°);

      2.2 жҜҸе®ҢжҲҗдёҖдёӘд»»еҠЎе°ұи°ғз”ЁcountDownLatch.countDown();пјҲд»»еҠЎж•°-1пјүиЎЁзӨәжӯӨйЎ№е°Ҹд»»еҠЎе·Іе®ҢжҲҗ

    3. еңЁдё»зәҝзЁӢи°ғз”ЁcountDownLatch.await();пјҲеҚізӣҙеҲ°д»»еҠЎж•°дёә0ж—¶пјүзӯүеҫ…жүҖжңүе°Ҹд»»еҠЎе®ҢжҲҗеҗҺпјҢеҶҚжү§иЎҢе…¶д»–ж“ҚдҪңгҖӮ

      еҪ“з„¶пјҢдёәйҳІжӯўдёҖзӣҙзӯүеҫ…пјҢеҸҜд»ҘеҠ дёӘи¶…ж—¶ж—¶й—ҙcountDownLatch.await(10L,TimeUnit.SECONDS);пјҢи¶…еҮәж—¶й—ҙеҗҺдёҚеҶҚзӯүеҫ…пјҢзӣҙжҺҘе”ӨйҶ’ await зәҝзЁӢиҝӣиЎҢдёӢдёҖжӯҘж“ҚдҪңгҖӮ

  • д»ҘдёҠжҖқи·Ҝе®һдҫӢд»Јз ҒеҸҜи§ҒеҸҰдёҖзҜҮpoiеӨҡзәҝзЁӢеӨҡsheetеҜјеҮәExcel,csv

# CountDownLatchи§Јжһҗ

CountDownLatch зұ»еӣҫеҰӮдёӢ:

д»ҺеӣҫдёӯеҸҜд»ҘзңӢеҮәпјҢCountDownLatch еҶ…йғЁдҫқиө– Sync е®һзҺ°пјҢиҖҢ Sync з»§жүҝиҮӘ AQSгҖӮCountDownLatch д»…жҸҗдҫӣдәҶдёҖдёӘжһ„йҖ ж–№жі•пјҡ

public CountDownLatch(int count) {
    if (count < 0) throw new IllegalArgumentException("count < 0");

    this.sync = new Sync(count);
}

private static final class Sync extends AbstractQueuedSynchronizer {

    private static final long serialVersionUID = 4982264981922014374L;

    // и®ҫзҪ®еҗҢжӯҘзҠ¶жҖҒеҖјпјҲй”Ғж Үи®°пјү
    Sync(int count) {
        setState(count);
    }

    // иҺ·еҸ–еҗҢжӯҘзҠ¶жҖҒеҖј
    int getCount() {
        return getState();
    }
    
    // е…ұдә«ејҸиҺ·еҸ–еҗҢжӯҘзҠ¶жҖҒ
    protected int tryAcquireShared(int acquires) {
        return (getState() == 0) ? 1 : -1;
    }

    // е…ұдә«ејҸйҮҠж”ҫеҗҢжӯҘзҠ¶жҖҒ
    protected boolean tryReleaseShared(int releases) {
        // Decrement count; signal when transition to zero
        for (;;) {
            // иҺ·еҸ–еҪ“еүҚstateеұһжҖ§зҡ„еҖј
            int c = getState();
            // еҰӮжһңstateдёә0пјҢеҲҷиҜҙжҳҺеҪ“еүҚи®Ўж•°еҷЁе·Із»Ҹи®Ўж•°е®ҢжҲҗпјҢзӣҙжҺҘиҝ”еӣһ
            if (c == 0)
                return false;
            int nextc = c-1;
            // дҪҝз”ЁCASз®—жі•еҜ№stateиҝӣиЎҢи®ҫзҪ®
            if (compareAndSetState(c, nextc))
                // и®ҫзҪ®жҲҗеҠҹеҗҺиҝ”еӣһеҪ“еүҚжҳҜеҗҰдёәжңҖеҗҺдёҖдёӘи®ҫзҪ®stateзҡ„зәҝзЁӢ
                return nextc == 0;
        }
    }

}

# await () ж–№жі•

public final void acquireSharedInterruptibly(int arg)
        throws InterruptedException {

    if (Thread.interrupted())
        throw new InterruptedException();

    if (tryAcquireShared(arg) < 0)
        doAcquireSharedInterruptibly(arg);
}

private void doAcquireSharedInterruptibly(int arg) throws InterruptedException {
  final Node node = addWaiter(Node.SHARED); // дҪҝз”ЁеҪ“еүҚзәҝзЁӢеҲӣе»әдёҖдёӘе…ұдә«жЁЎејҸзҡ„иҠӮзӮ№
  boolean failed = true;
  try {
    for (;;) {
      final Node p = node.predecessor();    // иҺ·еҸ–еҪ“еүҚиҠӮзӮ№зҡ„еүҚдёҖдёӘиҠӮзӮ№
      if (p == head) {  // еҲӨж–ӯеүҚдёҖдёӘиҠӮзӮ№жҳҜеҗҰдёәеӨҙз»“зӮ№
        int r = tryAcquireShared(arg);  // жҹҘзңӢеҪ“еүҚзәҝзЁӢжҳҜеҗҰиҺ·еҸ–еҲ°дәҶжү§иЎҢжқғйҷҗ
        if (r >= 0) {   // еӨ§дәҺ0иЎЁзӨәиҺ·еҸ–дәҶжү§иЎҢжқғйҷҗ
          
          /* еҸ‘зҺ° r еӨ§дәҺ 0пјҲеӣ дёә state е·Із»Ҹиў«зҪ®дёә 0 дәҶпјүпјҢ
           * иҜҘзәҝзЁӢе°ұдјҡи°ғз”Ё setHeadAndPropagate (Node, int) ж–№жі•пјҢ
           * е№¶дё”йҖҖеҮәеҪ“еүҚеҫӘзҺҜпјҢ
           * д№ҹе°ұејҖе§Ӣжү§иЎҢжҲ‘д»¬и°ғз”Ё await() ж–№жі•д№ӢеҗҺзҡ„д»Јз Ғ
           */
          setHeadAndPropagate(node, r); // е°ҶеҪ“еүҚиҠӮзӮ№и®ҫзҪ®дёәеӨҙз»“зӮ№пјҢе№¶дё”е”ӨйҶ’еҗҺйқўеӨ„дәҺзӯүеҫ…зҠ¶жҖҒзҡ„иҠӮзӮ№
          p.next = null; // help GC
          failed = false;
          return;
        }
      }
      
      // иө°еҲ°иҝҷдёҖжӯҘиҜҙжҳҺжІЎжңүиҺ·еҸ–еҲ°жү§иЎҢжқғйҷҗпјҢе°ұдҪҝеҪ“еүҚзәҝзЁӢиҝӣе…ҘвҖңжҗҒзҪ®вҖқзҠ¶жҖҒ
      if (shouldParkAfterFailedAcquire(p, node) &&
          parkAndCheckInterrupt())
        throw new InterruptedException();
    }
  } finally {
    if (failed)
      cancelAcquire(node);
  }
}

private void setHeadAndPropagate(Node node, int propagate) {
  Node h = head;
  setHead(node);    // е°ҶеҪ“еүҚиҠӮзӮ№и®ҫзҪ®дёәеӨҙиҠӮзӮ№
  // жЈҖжҹҘе”ӨйҶ’иҝҮзЁӢжҳҜеҗҰйңҖиҰҒеҫҖдёӢдј йҖ’пјҢе№¶дё”жЈҖжҹҘеӨҙз»“зӮ№зҡ„зӯүеҫ…зҠ¶жҖҒ
  if (propagate > 0 || h == null || h.waitStatus < 0 ||
      (h = head) == null || h.waitStatus < 0) {
    Node s = node.next;
    if (s == null || s.isShared())  // еҰӮжһңдёӢдёҖдёӘиҠӮзӮ№жҳҜе°қиҜ•д»Ҙе…ұдә«зҠ¶жҖҒиҺ·еҸ–иҺ·еҸ–жү§иЎҢжқғйҷҗзҡ„иҠӮзӮ№пјҢеҲҷе°Ҷе…¶е”ӨйҶ’
      doReleaseShared();
  }
}

protected boolean tryReleaseShared(int releases) {
    // Decrement count; signal when transition to zero
    for (;;) {
        // иҺ·еҸ–еҗҢжӯҘзҠ¶жҖҒ
        int c = getState();
        // еҗҢжӯҘзҠ¶жҖҒдёә0пјҢеҲҷзӣҙжҺҘиҝ”еӣһ
        if (c == 0)
            return false;
        // и®Ўз®—е№¶жӣҙж–°еҗҢжӯҘзҠ¶жҖҒ
        int nextc = c-1;
        if (compareAndSetState(c, nextc))
            return nextc == 0;
    }
}

private void doReleaseShared() {
  for (;;) {
    Node h = head;  // и®°еҪ•зӯүеҫ…йҳҹеҲ—дёӯзҡ„еӨҙз»“зӮ№зҡ„зәҝзЁӢ
    if (h != null && h != tail) {   // еӨҙз»“зӮ№дёҚдёәз©әпјҢдё”еӨҙз»“зӮ№дёҚзӯүдәҺе°ҫиҠӮзӮ№
      int ws = h.waitStatus;
      if (ws == Node.SIGNAL) {  // SIGNALзҠ¶жҖҒиЎЁзӨәеҪ“еүҚиҠӮзӮ№жӯЈеңЁзӯүеҫ…иў«е”ӨйҶ’
        if (!compareAndSetWaitStatus(h, Node.SIGNAL, 0))    // жё…йҷӨеҪ“еүҚиҠӮзӮ№зҡ„зӯүеҫ…зҠ¶жҖҒ
          continue;
        unparkSuccessor(h); // е”ӨйҶ’еҪ“еүҚиҠӮзӮ№зҡ„дёӢдёҖдёӘиҠӮзӮ№
      } else if (ws == 0 && !compareAndSetWaitStatus(h, 0, Node.PROPAGATE))
        continue;
    }
    // еҰӮжһңhиҝҳжҳҜжҢҮеҗ‘еӨҙз»“зӮ№пјҢиҜҙжҳҺеүҚйқўиҝҷж®өд»Јз Ғжү§иЎҢиҝҮзЁӢдёӯжІЎжңүе…¶д»–зәҝзЁӢеҜ№еӨҙз»“зӮ№иҝӣиЎҢиҝҮеӨ„зҗҶ
    if (h == head)  
      break;
  }
}
