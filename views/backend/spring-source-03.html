# еңЁ Bean зҡ„з”ҹе‘Ҫе‘Ёжңҹж·»еҠ жү©еұ•зӮ№

# е®ҡд№үдёҖдәӣиҒҢиҙЈе’ҢиғҪеҠӣпјҡ

  1. BeanFactoryPostProcessorпјҡз”ЁдәҺдҝ®ж”№ Spring е®№еҷЁдёӯ BeanDefinition зҡ„жү©еұ•зӮ№гҖӮ

    еҪ“ Spring е®№еҷЁеҲӣе»ә BeanDefinition д№ӢеҗҺпјҢдҪҶеңЁеҲӣе»ә Bean е®һдҫӢд№ӢеүҚпјҢBeanFactoryPostProcessor е°ұеҸҜд»ҘйҖҡиҝҮе®һзҺ° postProcessBeanFactory ж–№жі•жқҘдҝ®ж”№ BeanDefinition зҡ„еұһжҖ§пјҢеҰӮдҝ®ж”№ Bean зҡ„дҪңз”ЁеҹҹгҖҒж·»еҠ  Bean зҡ„еұһжҖ§еҖјзӯүгҖӮBeanFactoryPostProcessor зҡ„е®һзҺ°зұ»еңЁ Spring е®№еҷЁеҗҜеҠЁж—¶е°ұдјҡиў«и°ғз”ЁгҖӮ

  2. BeanPostProcessorпјҡз”ЁдәҺдҝ®ж”№ Bean е®һдҫӢзҡ„жү©еұ•зӮ№гҖӮ

    еҪ“ Spring е®№еҷЁеҲӣе»ә Bean е®һдҫӢд№ӢеҗҺпјҢдҪҶеңЁеҲқе§ӢеҢ– Bean д№ӢеүҚпјҢBeanPostProcessor е°ұеҸҜд»ҘйҖҡиҝҮе®һзҺ° postProcessBeforeInitialization е’Ң postProcessAfterInitialization ж–№жі•жқҘдҝ®ж”№ Bean зҡ„еұһжҖ§жҲ–жү§иЎҢдёҖдәӣеҲқе§ӢеҢ–ж“ҚдҪңгҖӮдҫӢеҰӮпјҢеҸҜд»Ҙдёә Bean ж·»еҠ д»ЈзҗҶеҜ№иұЎгҖҒдҝ®ж”№еұһжҖ§еҖјгҖҒжү§иЎҢ AOP ж“ҚдҪңзӯүгҖӮBeanPostProcessor зҡ„е®һзҺ°зұ»еңЁ Spring е®№еҷЁеҗҜеҠЁж—¶е°ұдјҡиў«и°ғз”ЁпјҢе№¶дё”дјҡеҜ№е®№еҷЁдёӯзҡ„жҜҸдёӘ Bean е®һдҫӢйғҪиҝӣиЎҢеӨ„зҗҶгҖӮ

  3. HierarchicalBeanFactoryпјҡжү©еұ• BeanFactory жҺҘеҸЈпјҢж”ҜжҢҒзҲ¶еӯҗе®№еҷЁпјҢиҝҷйҮҢжҡӮж—¶ж— з”ЁгҖӮ

  4. DefaultSingletonBeanRegistryпјҡ implements SingletonBeanRegistry

    е®һзҺ°дәҶж·»еҠ е’ҢиҺ·еҸ–еҚ•дҫӢ Bean зҡ„иғҪеҠӣ

  5. ConfigurableBeanFactoryпјҡextends HierarchicalBeanFactory, SingletonBeanRegistry

    жү©еұ• BeanFactory жҺҘеҸЈпјҢж”ҜжҢҒи®ҫзҪ®зҲ¶зә§е®№еҷЁгҖҒиҮӘиҮӘе®ҡд№үдҪңз”ЁеҹҹгҖҒж·»еҠ  BeanPostProcessor еҗҺзҪ®еӨ„зҗҶеҷЁгҖҒи®ҫзҪ® bean зҡ„еҲқе§ӢеҢ–е’Ңй”ҖжҜҒеӣһи°ғж–№жі•зӯүгҖӮ

  6. AutowireCapableBeanFactoryпјҡжү©еұ• BeanFactory жҺҘеҸЈпјҢж”ҜжҢҒиҮӘеҠЁиЈ…й…Қ Bean зҡ„иғҪеҠӣпјҢ

    иҝҷйҮҢдё»иҰҒж·»еҠ жү©еұ•зӮ№ Object applyBeanPostProcessorsBeforeInitialization(Object existingBean, String beanName); дҪңдёә еҲқе§ӢеҢ–еүҚ bean зҡ„еүҚзҪ®еӨ„зҗҶпјҢObject applyBeanPostProcessorsAfterInitialization(Object existingBean, String beanName); дҪңдёә еҲқе§ӢеҢ–еҗҺ bean зҡ„еҗҺзҪ®еӨ„зҗҶгҖӮ

# е…·дҪ“зҡ„е®һзҺ°пјҡ

  1. AbstractBeanFactoryпјҡ[дё»иҰҒе®һзҺ°дәҶdoGetBean()пјҢж·»еҠ BeanPostProcessor]

    extends DefaultSingletonBeanRegistryпјҲе·Із»Ҹе®һзҺ°дәҶж·»еҠ е’ҢиҺ·еҸ–еҚ•дҫӢ Bean зҡ„иғҪеҠӣпјү

    implements ConfigurableBeanFactoryпјҲж·»еҠ  BeanPostProcessor еҗҺзҪ®еӨ„зҗҶеҷЁпјү

    е®һзҺ°пјҡеҶ…зҪ®дёҖдёӘ List<BeanPostProcessor> еҲ—иЎЁпјҢе®һзҺ°дәҶ ж·»еҠ  BeanPostProcessor е’Ң иҺ·еҸ– BeanPostProcessor зҡ„иғҪеҠӣгҖӮ

  2. AbstractAutowireCapableBeanFactoryпјҡ[зңҹе®һзҡ„doGetBean()и°ғз”Ёзҡ„createBean(beanName, beanDefinition, args);зҡ„иҝҮзЁӢеңЁиҝҷдёӘзұ»йҮҢпјҢжүҖд»Ҙе°ҶBeanзҡ„жү©еұ•зӮ№еҠ е…ҘеҲ°еҲӣе»әBeanзҡ„иҝҮзЁӢдёӯ]

    extends AbstractBeanFactory пјҲе·Із»Ҹе…·еӨҮдәҶж·»еҠ BeanPostProcessorзҡ„иғҪеҠӣпјү

    implements AutowireCapableBeanFactoryпјҲе®һзҺ° bean зҡ„е®һдҫӢеҢ–еүҚеҗҺзҪ®еӣһи°ғпјү

    е®һзҺ°пјҡеңЁ createBean() дёӯпјҢдёҖдёӘ Bean е®һдҫӢеҢ–еҗҺиҝӣиЎҢеұһжҖ§еЎ«е……пјҢд№ӢеҗҺеҠ е…Ҙ Bean зҡ„ init()пјҢеңЁ init() еүҚеҗҺеҠ е…Ҙ BeanPostProcessor еүҚеҗҺзҪ®ж–№жі•зҡ„еӣһи°ғпјҢиҝҷж ·еңЁеҲӣе»әдёҖдёӘ Bean зҡ„ж—¶еҖҷе°ұеҠ е…ҘдәҶ Bean зҡ„з”ҹе‘Ҫе‘Ёжңҹе№Ійў„гҖӮ

  3. еҜ№дәҺ BeanFactoryPostProcessorпјҢеңЁжөӢиҜ•зҡ„ж—¶еҖҷжҲ‘д»¬жүӢеҶҷдёҖдёӘе®һзҺ°дәҶпјҢжүӢеҠЁи°ғз”ЁдёҖдёӢеҚіеҸҜгҖӮпјҲеҗҺз»ӯеҠ е…ҘдёҠдёӢж–Үзҡ„ж—¶еҖҷпјҢеңЁдёҠдёӢж–ҮйҮҢе®һзҺ°иҮӘеҠЁи°ғз”Ёпјү

# е°Ҷе®һзҺ°еҠ е…ҘеҺҹжңүзҡ„йҖ»иҫ‘дёӯпјҡ

  1. еҰӮдёҠ AbstractBeanFactory е’Ң AbstractAutowireCapableBeanFactory

public class CatBeanPostProcessor implements BeanPostProcessor {

    @Override
    public Object postProcessBeforeInitialization(Object bean, String beanName) {
        System.out.println("beanNameпјҡ" + beanName + " иҝӣе…ҘеҲ° Bean жү©еұ•зӮ№ BeanPostProcessor -> postProcessBeforeInitialization");
        if ("cat".equals(beanName)) {
            Cat cat = (Cat) bean;
            return new Cat("postProcessBeforeInitialization name cat");
        }
        return bean;
    }

    @Override
    public Object postProcessAfterInitialization(Object bean, String beanName) {
        System.out.println("beanNameпјҡ" + beanName + " иҝӣе…ҘеҲ° Bean жү©еұ•зӮ№ BeanPostProcessor -> postProcessAfterInitialization");
        return bean;
    }

}

public class DogBeanFactoryPostProcessor implements BeanFactoryPostProcessor {

    @Override
    public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) {
        System.out.println("иҝӣе…ҘеҲ° BeanDefinition жү©еұ•зӮ№ BeanFactoryPostProcessor -> postProcessBeanFactory");
        String beanName = "dog";
        BeanDefinition beanDefinition = beanFactory.getBeanDefinition(beanName);
        PropertyValues propertyValues = beanDefinition.getPropertyValues();
        PropertyValue propertyValue = propertyValues.getPropertyValue("name");
        String oldValue = String.valueOf(propertyValue.getValue());
        String updateValue = beanName + "-BeanDefinition-updated";
        propertyValue.setValue(updateValue);
        System.out.println("beanNameпјҡ" + beanName + " еұһжҖ§пјҡnameпјҢеҺҹе…ҲеҖјпјҡ" + oldValue + " дҝ®ж”№еҗҺеҖјпјҡ" + updateValue);
        System.out.println();
    }

}
