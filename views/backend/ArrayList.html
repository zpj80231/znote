# ArrayListеә•еұӮз®Җд»Ӣ

  • ArrayListжҳҜListжҺҘеҸЈзҡ„дёҖдёӘеҸҜеҸҳеӨ§е°Ҹзҡ„ж•°з»„зҡ„е®һзҺ°
  • ArrayListзҡ„еҶ…йғЁжҳҜдҪҝз”ЁдёҖдёӘObjectеҜ№иұЎж•°з»„жқҘеӯҳеӮЁе…ғзҙ зҡ„
  • еҲқе§ӢеҢ–ArrayListзҡ„ж—¶еҖҷпјҢеҸҜд»ҘжҢҮе®ҡеҲқе§ӢеҢ–е®№йҮҸзҡ„еӨ§е°ҸпјҢеҰӮжһңдёҚжҢҮе®ҡпјҢе°ұдјҡдҪҝз”Ёй»ҳи®ӨеӨ§е°ҸпјҢдёә10
  • еҪ“ж·»еҠ дёҖдёӘж–°е…ғзҙ зҡ„ж—¶еҖҷпјҢйҰ–е…ҲдјҡжЈҖжҹҘе®№йҮҸжҳҜеҗҰи¶іеӨҹж·»еҠ иҝҷдёӘе…ғзҙ пјҢеҰӮжһңеӨҹе°ұзӣҙжҺҘж·»еҠ пјҢеҰӮжһңдёҚеӨҹе°ұиҝӣиЎҢжү©е®№пјҢжү©е®№дёәеҺҹж•°з»„е®№йҮҸзҡ„1.5еҖҚпјҲ1.7д»ҘеҗҺпјү
  • еҪ“еңЁindexеӨ„ж”ҫзҪ®дёҖдёӘе…ғзҙ зҡ„ж—¶еҖҷпјҢдјҡе°Ҷж•°з»„indexеӨ„еҸіиҫ№зҡ„е…ғзҙ е…ЁйғЁеҸіз§»
  • еҪ“еңЁindexеӨ„еҲ йҷӨдёҖдёӘе…ғзҙ зҡ„ж—¶еҖҷпјҢдјҡе°Ҷж•°з»„indexеӨ„еҸіиҫ№зҡ„е…ғзҙ е…ЁйғЁе·Ұз§»

# жүӢеҶҷдёҖдёӘз®ҖеҚ•зҡ„MyArrayList

package com.zpj.electric.SourceCodeAnalysis;

/**
 * Created by admin on 2017/6/10.
 */
public class MyArrayList {

    //ArrayListжңүдёҖдёӘдё“й—Ёз”ЁжқҘиЈ…е…ғзҙ зҡ„е®№еҷЁпјҢдёәдәҶдҝқиҜҒйӣҶеҗҲд»Җд№Ҳзұ»еһӢзҡ„ж•°жҚ®йғҪеҸҜеӯҳеӮЁпјҢжүҖд»Ҙе®ҡд№үзҡ„Object[]
    public Object[] data;
    //ArrayListйҮҢиҝҳжңүдёҖдёӘеұһжҖ§пјҢз”ЁжқҘи®°еҪ•йӣҶеҗҲе…ғзҙ зҡ„дёӘж•°
    public int size;

    /**
     * жңүеҸӮжһ„йҖ ж–№жі•
     *
     * @param x жҢҮе®ҡж•°з»„еӨ§е°Ҹ
     */
    public MyArrayList(int x) {
        if (x > 0) {
            data = new Object[x];
        } else {
            System.out.println("еҸӮж•°ејӮеёё");
        }
    }

    /**
     * еҰӮжһңжҢҮе®ҡж•°з»„еӨ§е°ҸпјҢй»ҳи®Ө10дёӘ
     */
    public MyArrayList() {
        this(10);
    }

    /**
     * еҫ—еҲ°йӣҶеҗҲзҡ„еӨ§е°Ҹ  еҰӮпјҡint x = list.size();
     *
     * @return
     */
    public int size() {
        return size;
    }

    /**
     * еҫҖйӣҶеҗҲдёӯж·»еҠ е…ғзҙ  еҰӮпјҡlist.add(Object obj);
     *
     * @param obj
     */
    public void add(Object obj) {
        //еҪ“жҲ‘д»¬еҫҖйӣҶеҗҲдёӯж·»еҠ е…ғзҙ зҡ„ж—¶еҖҷпјҢobjжңҖз»ҲйғҪдјҡж·»еҠ иҝӣObject[]ж•°з»„дёӯеҺ»
        //жүҖд»ҘжҜҸж¬Ўж·»еҠ ж•°жҚ®зҡ„ж—¶еҖҷйғҪйңҖиҰҒеҲӨж–ӯObject[],еҚіdataж•°з»„жңүжІЎжңүеЎ«ж»Ў
        if (data.length == size) {
            //еҰӮжһңеЎ«ж»ЎдәҶпјҢйӮЈд№ҲйңҖиҰҒжү©е®№:
            // jdk1.7д№ӢеүҚпјҡsize*3/2+1
            // jdk1.7еҸҠд№ӢеҗҺпјҡsize+(size>>1)
            Object[] temp = new Object[size + (size >> 1)];
            //жү©е®№еҗҺе°ҶиҖҒж•°з»„еӨҚеҲ¶еҲ°ж–°ж•°з»„йҮҢ
            System.arraycopy(data, 0, temp, 0, size);
            //ж”№еҸҳеј•з”ЁжҢҮеҗ‘ gcеӣһж”¶иҖҒж•°з»„еҜ№иұЎ
            data = temp;
        }
        
        data[size] = obj;
        size++;
    }

    /**
     * жҢүжҢҮе®ҡдёӢж ҮеҲ йҷӨйӣҶеҗҲдёӯзҡ„е…ғзҙ 
     *
     * @param index
     */
    public void remove(int index) {
        //System.arraycopy(иҰҒиў«еӨҚеҲ¶зҡ„иҖҒж•°з»„пјҢд»ҺдёӢж ҮindexејҖе§ӢеӨҚеҲ¶пјҢиҰҒеӨҚеҲ¶еҲ°зҡ„ж–°ж•°з»„пјҢд»Һж–°ж•°з»„зҡ„дёӢж ҮindexжҸ’е…Ҙ,д»ҺиҖҒж•°з»„дёӢж ҮејҖе§ӢиҰҒиў«еӨҚеҲ¶зҡ„дёӘж•°);
        System.arraycopy(data, index + 1, data, index, size - (index + 1));
        size--;
    }

    /**
     * жҢҮе®ҡе…ғзҙ еҲ йҷӨйӣҶеҗҲдёӯзҡ„е…ғзҙ 
     *
     * @param obj
     */
    public void remove(Object obj) {
        //жҜҸеҪ“жҢҮе®ҡе…ғзҙ еҲ йҷӨзҡ„ж—¶еҖҷпјҢеә•еұӮдјҡжӢҝзқҖobjе’ҢжҜҸдёӘе…ғзҙ еҒҡequalsжҜ”иҫғ
        for (int x = 0; x < size; x++) {
            if (obj.equals(data[x])) {
                remove(x);//жҢүдёӢж ҮеҲ йҷӨе…ғзҙ 
                break;//дёҖдёӘremoveж–№жі•еҸӘиғҪеҲ йҷӨдёҖдёӘеҜ№иұЎ
            }
        }
    }

    /**
     * ж №жҚ®жҢҮе®ҡдёӢж ҮиҺ·еҫ—е…ғзҙ 
     *
     * @param x
     * @return
     */
    public Object get(int x) {
        return x >= 0 && x < size ? data[x] : "еҸӮж•°и¶Ҡз•ҢејӮеёё";
    }

}

# ArrayListзҡ„зәҝзЁӢе®үе…ЁжҖ§

еҜ№ArrayListиҝӣиЎҢж·»еҠ е…ғзҙ зҡ„ж“ҚдҪңзҡ„ж—¶еҖҷжҳҜеҲҶдёӨдёӘжӯҘйӘӨиҝӣиЎҢзҡ„пјҢеҚіз¬¬дёҖжӯҘе…ҲеңЁobject[size]зҡ„дҪҚзҪ®дёҠеӯҳж”ҫйңҖиҰҒж·»еҠ зҡ„е…ғзҙ пјӣз¬¬дәҢжӯҘе°Ҷsizeзҡ„еҖјеўһеҠ 1гҖӮз”ұдәҺиҝҷдёӘиҝҮзЁӢеңЁеӨҡзәҝзЁӢзҡ„зҺҜеўғдёӢжҳҜдёҚиғҪдҝқиҜҒе…·жңүеҺҹеӯҗжҖ§зҡ„пјҢеӣ жӯӨArrayListеңЁеӨҡзәҝзЁӢзҡ„зҺҜеўғдёӢжҳҜзәҝзЁӢдёҚе®үе…Ёзҡ„гҖӮ

е…·дҪ“дёҫдҫӢиҜҙжҳҺпјҡеңЁеҚ•зәҝзЁӢиҝҗиЎҢзҡ„жғ…еҶөдёӢпјҢеҰӮжһңSize = 0пјҢж·»еҠ дёҖдёӘе…ғзҙ еҗҺпјҢжӯӨе…ғзҙ еңЁдҪҚзҪ® 0пјҢиҖҢдё”Size=1пјӣиҖҢеҰӮжһңжҳҜеңЁеӨҡзәҝзЁӢжғ…еҶөдёӢпјҢжҜ”еҰӮжңүдёӨдёӘзәҝзЁӢпјҢзәҝзЁӢ A е…Ҳе°Ҷе…ғзҙ еӯҳж”ҫеңЁдҪҚзҪ®0гҖӮдҪҶжҳҜжӯӨж—¶ CPU и°ғеәҰзәҝзЁӢAжҡӮеҒңпјҢзәҝзЁӢ B еҫ—еҲ°иҝҗиЎҢзҡ„жңәдјҡгҖӮзәҝзЁӢBд№ҹеҗ‘жӯӨArrayList ж·»еҠ е…ғзҙ пјҢеӣ дёәжӯӨж—¶ Size д»Қз„¶зӯүдәҺ 0 пјҲжіЁж„Ҹе“ҰпјҢжҲ‘д»¬еҒҮи®ҫзҡ„жҳҜж·»еҠ дёҖдёӘе…ғзҙ жҳҜиҰҒдёӨдёӘжӯҘйӘӨе“ҰпјҢиҖҢзәҝзЁӢAд»…д»…е®ҢжҲҗдәҶжӯҘйӘӨ1пјүпјҢжүҖд»ҘзәҝзЁӢBд№ҹе°Ҷе…ғзҙ еӯҳж”ҫеңЁдҪҚзҪ®0гҖӮз„¶еҗҺзәҝзЁӢAе’ҢзәҝзЁӢBйғҪз»§з»ӯиҝҗиЎҢпјҢйғҪеўһ еҠ  Size зҡ„еҖјгҖӮ йӮЈеҘҪпјҢзҺ°еңЁжҲ‘д»¬жқҘзңӢзңӢ ArrayList зҡ„жғ…еҶөпјҢе…ғзҙ е®һйҷ…дёҠеҸӘжңүдёҖдёӘпјҢеӯҳж”ҫеңЁдҪҚзҪ® 0пјҢиҖҢSizeеҚҙзӯүдәҺ 2гҖӮиҝҷе°ұжҳҜвҖңзәҝзЁӢдёҚе®үе…ЁвҖқдәҶгҖӮ

еҰӮжһңйқһиҰҒеңЁеӨҡзәҝзЁӢзҡ„зҺҜеўғдёӢдҪҝз”ЁArrayListпјҢе°ұйңҖиҰҒдҝқиҜҒе®ғзҡ„зәҝзЁӢе®үе…ЁжҖ§пјҢйҖҡеёёжңүдёӨз§Қи§ЈеҶіеҠһжі•пјҡз¬¬дёҖпјҢдҪҝз”Ёsynchronizedе…ій”®еӯ—пјӣз¬¬дәҢпјҢеҸҜд»Ҙз”ЁCollectionsзұ»дёӯзҡ„йқҷжҖҒж–№жі•synchronizedList();еҜ№ArrayListиҝӣиЎҢи°ғз”ЁеҚіеҸҜгҖӮ
